/* to find sub array in given array*/
#include<stdio.h>
#define max_size 100
int subarray(int *ptr,int *ptr1,int size,int size1)
{
    int i,j,res=0;
    for(i=0;i<size;i++)
    {
        if(ptr[i] == ptr1[0])
        {
            for(j=0;j<size1;j++)
            {
                if(ptr[i+j] != ptr1[j])
                {
                    res = 0;
                    break;
                }
            }
        }
        if(j == size1) 
        res = 1;
    }
    return res;
}
int main()
{
    int arr[max_size]={1,2,3,4,5,6};
    int *ptr=arr;
    int arr1[]={1,2,3};
    int *ptr1=arr1;
    int size=sizeof(arr)/sizeof(arr[0]);
    int size1=sizeof(arr1)/sizeof(arr1[0]);
    if(subarray(ptr,ptr1,size,size1))
    {
        printf("sub array\n");
    }
    else
    {
        printf("not subarray\n");
    }
}
